    async function onConfigChange(config) {
      const bgColor = config.background_color || defaultConfig.background_color;
      const cardColor = config.card_color || defaultConfig.card_color;
      const textColor = config.text_color || defaultConfig.text_color;
      const accentColor = config.accent_color || defaultConfig.accent_color;

      const fontSize = config.font_size || defaultConfig.font_size;

      document.body.style.fontFamily = `${fontFamily}, -apple-system, BlinkMacSystemFont, 'Segoe UI', sans-serif`;
      document.body.style.fontSize = `${fontSize}px`;
      
      const gradientBg = document.querySelector('.gradient-bg');
      if (gradientBg) {
        gradientBg.style.background = `linear-gradient(135deg, ${bgColor} 0%, ${accentColor} 100%)`;
      }

      const cards = document.querySelectorAll('.card');
      cards.forEach(card => {
        card.style.backgroundColor = cardColor;
      });

      const allText = document.querySelectorAll('h1, h2, p, span');
      allText.forEach(el => {
        if (!el.classList.contains('skill-badge')) {
          el.style.color = textColor;
        }
      });

      const name = config.student_name || defaultConfig.student_name;
      document.getElementById('name').textContent = name;
      
      const initials = name.split(' ').map(word => word[0]).join('').substring(0, 2).toUpperCase();
      document.getElementById('initial').textContent = initials;
      
      document.getElementById('class').textContent = config.student_class || defaultConfig.student_class;
      document.getElementById('school').textContent = config.school_name || defaultConfig.school_name;
      document.getElementById('age').textContent = config.student_age || defaultConfig.student_age;
      document.getElementById('about').textContent = config.about_text || defaultConfig.about_text;
      document.getElementById('education').textContent = config.education_info || defaultConfig.education_info;
      document.getElementById('contact').textContent = config.contact_info || defaultConfig.contact_info;

      const skillsContainer = document.getElementById('skills-container');
      skillsContainer.innerHTML = '';
      const skills = [
        config.skill_1 || defaultConfig.skill_1,
        config.skill_2 || defaultConfig.skill_2,
        config.skill_3 || defaultConfig.skill_3,
        config.skill_4 || defaultConfig.skill_4,
        config.skill_5 || defaultConfig.skill_5
      ].filter(skill => skill);

      skills.forEach(skill => {
        const badge = document.createElement('span');

        badge.style.background = `linear-gradient(to right, ${accentColor}, ${bgColor})`;

        badge.style.fontSize = `${fontSize}px`;
        badge.textContent = skill;
        skillsContainer.appendChild(badge);
      });

      const sectionTitles = document.querySelectorAll('.section-title');

        title.style.fontSize = `${fontSize * 1.5}px`;
      });

      document.getElementById('name').style.fontSize = `${fontSize * 2}px`;
    }
